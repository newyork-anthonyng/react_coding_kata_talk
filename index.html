              class Base extends Component {
                constructor() {
                  super();

                  this.state = {
                    online: navigator.onLine,
                  };
                  // ...
                }
                // No render method!
              }

              export class Online extends Base {
                render() {
                  return this.state.online ? this.renderChildren() : null;
                }
              }

              export class Offline extends Base {
                render() {
                  return !this.state.online ? this.renderChildren() : null;
                }
              }
